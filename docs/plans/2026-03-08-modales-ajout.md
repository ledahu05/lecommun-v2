# Modales d'ajout dépense/ajustement — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace inline Card forms with Dialog modals triggered from buttons on dashboard and list pages. Replace `<select>` with button grids for sous-catégories.

**Architecture:** shadcn/ui Dialog wraps existing form components (Card wrapper removed). New `DepenseDialog` and `AjustementDialog` client components manage open/close state and are reused on dashboard and list pages. Forms accept an `onSuccess` callback to close the dialog after submission.

**Tech Stack:** Next.js 15, React 19, shadcn/ui Dialog, Server Actions, Tailwind CSS 3

**Design doc:** `docs/plans/2026-03-08-modales-ajout-design.md`

Replace the entire `components/depenses/DepenseForm.tsx` with:

```tsx
'use client';

import { useState, useTransition } from 'react';
import { useRouter } from 'next/navigation';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { CATEGORIES } from '@/lib/categories';
import type { Categorie } from '@/lib/categories';
import { actionCreateDepense } from '@/app/(app)/depenses/actions';

const SOUS_CAT_LABELS: Record<string, string> = {
  marche_pain: 'Marché / Pain',
  intermarche: 'Intermarché',
  viande: 'Viande',
  miel: 'Miel',
  loyer: 'Loyer',
  energie: 'Énergie',
  internet: 'Internet',
  eau: 'Eau',
  assurance: 'Assurance',
  amazon: 'Amazon',
  netflix: 'Netflix',
  restaurant: 'Restaurant',
  vacances: 'Vacances',
  sport: 'Sport',
  livres: 'Livres',
  pharmacie: 'Pharmacie',
  animaux: 'Animaux',
  cadeaux: 'Cadeaux',
  appartement: 'Appartement',
};

interface DepenseFormProps {
  onSuccess?: () => void;
}

export default function DepenseForm({ onSuccess }: DepenseFormProps) {
  const [categorie, setCategorie] = useState<Categorie>('alimentation');
  const [sousCategorie, setSousCategorie] = useState<string>(
    CATEGORIES.alimentation.sous_categories[0]
  );
  const [payePar, setPayePar] = useState<'chris' | 'alex'>('chris');
  const [isPending, startTransition] = useTransition();
  const [error, setError] = useState<string | null>(null);
  const router = useRouter();

  const sousCats = CATEGORIES[categorie].sous_categories;

  function handleCategorieChange(cat: Categorie) {
    setCategorie(cat);
    setSousCategorie(CATEGORIES[cat].sous_categories[0]);
  }

  async function handleSubmit(formData: FormData): Promise<void> {
    setError(null);
    const result = await actionCreateDepense(formData);
    if (result && 'error' in result) {
      setError(result.error);
      return;
    }
    startTransition(() => {
      router.refresh();
      onSuccess?.();
    });
  }

  const today = new Date().toISOString().slice(0, 10);

  return (
    <form action={handleSubmit} data-testid="depense-form" className="space-y-4">
      {/* Catégorie — 4 boutons */}
      <div>
        <Label className="text-base mb-2 block">Catégorie</Label>
        <input type="hidden" name="categorie" value={categorie} />
        <div className="grid grid-cols-2 gap-2">
          {(Object.keys(CATEGORIES) as Categorie[]).map((cat) => (
            <Button
              key={cat}
              type="button"
              variant={categorie === cat ? 'default' : 'outline'}
              className="min-h-[48px] text-base"
              onClick={() => handleCategorieChange(cat)}
            >
              {CATEGORIES[cat].emoji} {CATEGORIES[cat].label}
            </Button>
          ))}
        </div>
      </div>

      {/* Sous-catégorie — boutons */}
      <div>
        <Label className="text-base mb-2 block">Sous-catégorie</Label>
        <input type="hidden" name="sous_categorie" value={sousCategorie} />
        <div className="flex flex-wrap gap-2">
          {sousCats.map((sc) => (
            <Button
              key={sc}
              type="button"
              variant={sousCategorie === sc ? 'default' : 'outline'}
              className="min-h-[48px] text-base"
              onClick={() => setSousCategorie(sc)}
            >
              {SOUS_CAT_LABELS[sc] ?? sc}
            </Button>
          ))}
        </div>
      </div>

      {/* Payeur — segmented buttons */}
      <div>
        <Label className="text-base mb-2 block">Payé par</Label>
        <input type="hidden" name="paye_par" value={payePar} />
        <div className="flex gap-2">
          <Button
            type="button"
            variant={payePar === 'chris' ? 'default' : 'outline'}
            className="flex-1 min-h-[48px] text-base"
            onClick={() => setPayePar('chris')}
          >
            Chris
          </Button>
          <Button
            type="button"
            variant={payePar === 'alex' ? 'default' : 'outline'}
            className="flex-1 min-h-[48px] text-base"
            onClick={() => setPayePar('alex')}
          >
            Alex
          </Button>
        </div>
      </div>

      {/* Montant */}
      <div>
        <Label htmlFor="montant" className="text-base mb-2 block">
          Montant (€)
        </Label>
        <Input
          id="montant"
          name="montant"
          type="number"
          min="0.01"
          step="0.01"
          placeholder="0,00"
          className="min-h-[48px] text-base"
          required
        />
      </div>

      {/* Date */}
      <div>
        <Label htmlFor="date_depense" className="text-base mb-2 block">
          Date
        </Label>
        <Input
          id="date_depense"
          name="date_depense"
          type="date"
          defaultValue={today}
          className="min-h-[48px] text-base"
          required
        />
      </div>

      {/* Libellé optionnel */}
      <div>
        <Label htmlFor="label" className="text-base mb-2 block">
          Libellé (optionnel)
        </Label>
        <Input
          id="label"
          name="label"
          type="text"
          placeholder="Libellé optionnel"
          className="min-h-[48px] text-base"
        />
      </div>

      {error && <p className="text-sm text-destructive">{error}</p>}

      <Button type="submit" className="w-full min-h-[48px] text-base" disabled={isPending}>
        {isPending ? 'Ajout...' : 'Ajouter'}
      </Button>
    </form>
  );
}
```

Key changes from original:
- No `Card`/`CardHeader`/`CardContent` wrapper
- `<select>` replaced with button grid for sous-catégories using `SOUS_CAT_LABELS` for display
- `sousCategorie` state tracked, resets when category changes
- Accepts `onSuccess` callback prop
- Uses `useTransition` + `router.refresh()` for post-submit refresh
- Shows error message inline
- Shows "Ajout..." pending state on submit button

import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,

interface DepenseDialogProps {

export function DepenseDialog({ trigger }: DepenseDialogProps) {
  const [open, setOpen] = useState(false);

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        {trigger ?? (
          <Button variant="outline" className="min-h-[48px] text-base">
            + Dépense
          </Button>
        )}
      </DialogTrigger>
      <DialogContent className="max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>Nouvelle dépense</DialogTitle>
        </DialogHeader>
        <DepenseForm onSuccess={() => setOpen(false)} />

}
