    ```typescript
    import { createClient } from '@libsql/client';

    // Client direct (pas Drizzle) pour les tests — évite les imports serveur Next.js
    function getTestClient() {
      const url = process.env.TURSO_DATABASE_URL;
      const authToken = process.env.TURSO_AUTH_TOKEN;
      if (!url) throw new Error('TURSO_DATABASE_URL manquant dans .env.local');
      return createClient({ url, authToken });
    }

    export interface SeedMois {
      id?: number;
      annee: number;
      mois: number;
      balance_reportee: number;
    }

    export interface SeedDepense {
      mois_id: number;
      categorie: string;
      sous_categorie: string;
      paye_par: 'chris' | 'alex';
      montant: number;
      label?: string;
      date_depense: string; // 'YYYY-MM-DD'
    }

    export interface SeedAjustement {
      mois_id: number;
      de: 'chris' | 'alex';
      vers: 'chris' | 'alex';
      montant: number;
      label: string;
      date_ajustement: string; // 'YYYY-MM-DD'
    }

    export interface SeedData {
      mois: SeedMois[];
      depenses?: SeedDepense[];
      ajustements?: SeedAjustement[];
    }

    export async function seedDatabase(data: SeedData): Promise<void> {
      const client = getTestClient();

      // Vider dans l'ordre correct (FK: depenses/ajustements avant mois)
      await client.execute('DELETE FROM depenses');
      await client.execute('DELETE FROM ajustements');
      await client.execute('DELETE FROM mois');

      // Insérer les mois
      for (const m of data.mois) {
        await client.execute({
          sql: 'INSERT INTO mois (annee, mois, balance_reportee, cree_le) VALUES (?, ?, ?, ?)',
          args: [m.annee, m.mois, m.balance_reportee, Math.floor(Date.now() / 1000)],
        });
      }

      // Insérer les dépenses
      for (const d of data.depenses ?? []) {
        const ts = Math.floor(new Date(d.date_depense).getTime() / 1000);
        await client.execute({
          sql: `INSERT INTO depenses (mois_id, categorie, sous_categorie, paye_par, montant, label, date_depense, cree_le)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
          args: [d.mois_id, d.categorie, d.sous_categorie, d.paye_par, d.montant, d.label ?? null, ts, Math.floor(Date.now() / 1000)],
        });
      }

      // Insérer les ajustements
      for (const a of data.ajustements ?? []) {
        const ts = Math.floor(new Date(a.date_ajustement).getTime() / 1000);
        await client.execute({
          sql: `INSERT INTO ajustements (mois_id, de, vers, montant, label, date_ajustement, cree_le)
                VALUES (?, ?, ?, ?, ?, ?, ?)`,
          args: [a.mois_id, a.de, a.vers, a.montant, a.label, ts, Math.floor(Date.now() / 1000)],
        });
      }

      await client.close();
    }
    ```

    Note: `seedDatabase` insère les mois sans ID explicite (laisse SQLite auto-incrémenter) pour éviter les conflits avec la contrainte unique. La table mois est vidée avant chaque seed — c'est correct pour les tests E2E.

    Note: Pour les scénarios de balance, les mois_id dans SeedDepense/SeedAjustement doivent correspondre aux mois insérés. En pratique, après un DELETE + INSERT, les IDs SQLite ne recommencent PAS toujours à 1 (AUTOINCREMENT). Pour les tests, utiliser des requêtes SELECT après l'INSERT des mois pour récupérer les vrais IDs — ou simplifier en insérant une dépense sans mois_id préalablement connu en utilisant une sous-requête. Alternative plus simple: ne pas passer de mois_id hardcodé dans SeedDepense mais calculer via le numéro de mois/année. Implémenter comme suit dans seedDatabase: après insertion des mois, récupérer leurs IDs réels via SELECT et mapper par (annee, mois). Mettre à jour le code de seed des dépenses/ajustements pour utiliser ces IDs réels.
